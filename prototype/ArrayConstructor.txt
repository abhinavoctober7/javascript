function MyArray(...args) {
  // Create empty object
  let arr = {};

  // Add length property
  Object.defineProperty(arr, "length", {
    value: 0,
    writable: true,
    enumerable: false
  });

  for (let i = 0; i < args.length; i++) {
      arr[i] = args[i];
  }
  arr.length = args.length;

  // Attach Array-like prototype
  Object.setPrototypeOf(arr, MyArray.prototype);

  return arr;
}

‚≠ê Add array methods in prototype (like push, pop)
MyArray.prototype.push = function (value) {
  this[this.length] = value;
  this.length++;
  return this.length;
};

MyArray.prototype.pop = function () {
  if (this.length === 0) return undefined;
  let value = this[this.length - 1];
  delete this[this.length - 1];
  this.length--;
  return value;
};
