‚≠ê Polyfill for call()
Function.prototype.myCall = function (context, ...args) {
  context = context || globalThis;   // default to global (window/global)
  const fn = Symbol();               // temporary unique key
  context[fn] = this;                // 'this' = function
  const result = context[fn](...args);
  delete context[fn];                // cleanup
  return result;
};

‚≠ê Polyfill for apply()
Function.prototype.myApply = function (context, argsArray = []) {
  context = context || globalThis;
  const fn = Symbol();
  context[fn] = this;
  const result = context[fn](...argsArray);
  delete context[fn];
  return result;
};

‚≠ê Polyfill for bind()
Function.prototype.myBind = function (context, ...args) {
    const fn = this;
    return function (...newArgs) {
        return fn.call(context, ...args, ...newArgs);
  };
};
